$(".youtube").colorbox({
iframe:true,
innerWidth:970,
innerHeight:600,
onOpen: function(){
$("#cboxClose").css("opacity",0);
},
onComplete: function(){
setTimeout(function(){
$("#cboxClose").css("opacity",1);
},2000);
}
});

$(".colorbox1").colorbox({

}
});

$(".colorbox2").colorbox({
rel:'colorbox2',

}
});
